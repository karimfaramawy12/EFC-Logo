        // --- STATE & APP LOGIC ---
        let lang = 'ar';
        let page = 'home';
        let isMenuOpen = false;

        function toggleLang() {
            lang = lang === 'ar' ? 'en' : 'ar';
            renderApp();
        }

        function setPage(newPage) {
            page = newPage;
            isMenuOpen = false; // close menu on nav
            window.scrollTo(0, 0);
            renderApp();
        }

        function toggleMenu() {
            isMenuOpen = !isMenuOpen;
            renderApp();
        }

        function getDir() {
            return CONTENT[lang].meta.dir;
        }

        // --- RENDER HELPERS ---
        
        function NavLink(to, label, isMobile) {
            const activeClass = page === to ? 'font-bold text-blue-200' : '';
            const mobileClasses = 'block w-full text-start py-3 px-4 border-b border-gray-100';
            const desktopClasses = 'mx-4 hover:text-blue-300 transition-colors';
            const baseClass = isMobile ? mobileClasses : desktopClasses;
            
            return `<button onclick="setPage('${to}')" class="${baseClass} ${activeClass}">${label}</button>`;
        }

        function renderApp() {
            const app = document.getElementById('app');
            
            // Update HTML Lang/Dir attributes
            document.documentElement.lang = lang;
            document.documentElement.dir = CONTENT[lang].meta.dir;
            
            // Update Body Font Class
            document.body.className = `bg-[#F6FBFE] text-[#000105] ${CONTENT[lang].meta.font}`;

            let pageContent = '';
            if (page === 'home') pageContent = renderHome();
            else if (page === 'about') pageContent = renderAbout();
            else if (page === 'services') pageContent = renderServices();

            app.innerHTML = `
                ${Header()}
                <main class="flex-grow">
                    ${pageContent}
                </main>
                ${Footer()}
            `;
            
            // Re-initialize Icons
            lucide.createIcons();
        }
